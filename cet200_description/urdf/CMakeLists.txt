# Generate URDF from XACRO in colcon build
find_program(xacro_EXECUTABLE xacro)

if(NOT xacro_EXECUTABLE)
  message(FATAL_ERROR "xacro not found.")
endif()

set(XACRO_FILE cet200.xacro)
set(URDF_FILE cet200.urdf)

set(XACRO_PATH ${CMAKE_CURRENT_SOURCE_DIR}/${XACRO_FILE})
set(URDF_PATH ${CMAKE_CURRENT_SOURCE_DIR}/${URDF_FILE})

add_custom_command(
  OUTPUT ${URDF_PATH}
  COMMAND ${xacro_EXECUTABLE} ${XACRO_PATH} -o ${URDF_PATH}
  COMMAND sed -i 's|autogenerated by xacro from .*|autogenerated by xacro from ${XACRO_FILE} -->|g' ${URDF_PATH}
  DEPENDS ${XACRO_PATH}
  COMMENT "Generating URDF from XACRO: ${URDF_PATH}"
)

add_custom_target(generate_urdf ALL
  DEPENDS ${URDF_PATH}
)
